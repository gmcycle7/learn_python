Python 自動化

本附錄頁面是你使用 Python 自動化日常任務的實用手冊：管理檔案、處理文字、 執行 shell 命令和建立命令列工具。每個範例都是完整、可執行的腳本， 你可以立即應用到自己的工作中。

給 MATLAB 使用者：在 MATLAB 中，你可能使用 system() 呼叫 shell 命令，或撰寫 .m 腳本來批次處理資料檔案。Python 以更豐富的標準函式庫提供相同的功能 — 不需要額外工具箱。如果你曾寫過 MATLAB 腳本來遍歷資料夾中的 檔案、清理資料並儲存結果，在這裡你會感到很熟悉。

1. 為什麼 Python 適合自動化

Python 已成為自動化腳本的首選語言，原因如下：

  • 可讀的語法 — 你今天寫的腳本，六個月後仍然容易理解。不需要晦澀的 bash 單行指令。
  • 跨平台 — 同一個 Python 腳本在 Windows、macOS 和 Linux 上無需修改即可執行（當你使用標準函式庫處理檔案路徑和系統呼叫時）。
  • 電池已充飽 — 標準函式庫附帶檔案 I/O、JSON、CSV、正規表達式、子程序、引數解析等模組 — 不需要 pip install。
  • 取代 bash 腳本 — 超過幾行的任務，Python 比 shell 腳本更易維護、更可移植、更容易除錯。
  • 易於擴展 — 需要更多功能時，數千個第三方套件只需一個 pip install 即可取得。
本頁使用的標準函式庫模組： os、os.path、pathlib、 shutil、glob、json、 csv、re、subprocess、 sys、argparse、time、 datetime。這些都隨每個 Python 安裝附帶。

2. 檔案系統操作

檔案管理是自動化的基礎。Python 提供兩種主要方法：較舊的 os / os.path 模組和現代的 pathlib 模組（建議新程式碼使用）。

2.1 列出目錄中的檔案

# --- 方法 1：os.listdir（回傳名稱的純串列）---
import os

entries = os.listdir(".")          # 目前目錄
print(entries)
# ['report.csv', 'data', 'script.py', ...]

# --- 方法 2：pathlib.Path.iterdir（回傳 Path 物件）---
from pathlib import Path

for item in Path(".").iterdir():
    print(item.name, "dir" if item.is_dir() else "file")

# --- 方法 3：glob（模式匹配）---
import glob

csv_files = glob.glob("*.csv")     # 目前目錄中所有 CSV
print(csv_files)

# pathlib 等價寫法
csv_files = list(Path(".").glob("*.csv"))
print(csv_files)

2.3 建立和刪除資料夾

import os
import shutil

# 建立巢狀資料夾結構（類似 mkdir -p）
os.makedirs("output/reports/2025", exist_ok=True)

# pathlib 等價寫法

Path("output/reports/2025").mkdir(parents=True, exist_ok=True)

2.7 完整範例：整理下載資料夾

可執行腳本！將此儲存為 organize_downloads.py，指向任何雜亂的資料夾。 它會按副檔名將檔案分類到子資料夾中。 
#!/usr/bin/env python3
"""將目錄中的檔案依副檔名整理到子資料夾。"""

import os
import shutil
from pathlib import Path

def organize_folder(folder_path):
    """將檔案移動到以副檔名命名的子資料夾中。"""
    folder = Path(folder_path)

    if not folder.is_dir():
        print(f"Error: {folder} is not a directory")
        return

    moved = 0
    for item in folder.iterdir():
        # 跳過目錄和隱藏檔案
        if item.is_dir() or item.name.startswith("."):
            continue

        # 決定目標子資料夾
        ext = item.suffix.lower()          # 例如 '.pdf'
        if ext == "":
            dest_folder = folder / "no_extension"
        else:
            dest_folder = folder / ext.lstrip(".")   # 'pdf', 'jpg' 等

        dest_folder.mkdir(exist_ok=True)
        dest_file = dest_folder / item.name

        # 避免覆寫已存在的檔案
        if dest_file.exists():
            print(f"  Skipped (already exists): {item.name}")
            continue

        shutil.move(str(item), str(dest_file))
        moved += 1
        print(f"  Moved: {item.name} -> {dest_folder.name}/")

    print(f"\nDone! Moved {moved} file(s).")

if __name__ == "__main__":
    import sys
    target = sys.argv[1] if len(sys.argv) > 1 else "."
    organize_folder(target)
# 用法：
# python organize_downloads.py ~/Downloads
#   Moved: report.pdf -> pdf/
#   Moved: photo.jpg -> jpg/
#   Moved: notes.txt -> txt/
#   Done! Moved 3 file(s).

3. 處理文字檔案

3.1 讀取、寫入和附加

# --- 讀取整個檔案 ---
with open("notes.txt", "r") as f:
    content = f.read()
print(content)

# --- 逐行讀取（適合大檔案，節省記憶體）---
with open("notes.txt", "r") as f:
    for line in f:
        print(line.strip())        # strip() 移除尾部的換行符

# --- 寫入（覆蓋現有內容）---
with open("output.txt", "w") as f:
    f.write("Line 1\n")
    f.write("Line 2\n")

# --- 附加（添加到末尾）---
with open("output.txt", "a") as f:
    f.write("Line 3 (appended)\n")

3.3 處理 JSON 檔案

import json

# --- 讀取 JSON ---
with open("config.json", "r") as f:
    config = json.load(f)          # 回傳字典（或串列）
print(config["database"]["host"])

# --- 寫入 JSON ---
data = {
    "name": "AutoBot",
    "version": 2,
    "features": ["file_ops", "scheduling"],
}
with open("output.json", "w") as f:
    json.dump(data, f, indent=2)   # indent 讓輸出更易讀

# --- 處理 JSON 字串 ---
json_string = '{"x": 1, "y": 2}'
obj = json.loads(json_string)      # loads = 從字串載入
print(obj["x"])                    # 1

back_to_string = json.dumps(obj)
print(back_to_string)              # '{"x": 1, "y": 2}'
提示：json.load() 從檔案讀取。 json.loads() 從字串讀取。 輸出也是同樣的模式：json.dump() 寫入檔案， json.dumps() 回傳字串。尾部的 s 代表 string（字串）。

3.5 使用正規表達式搜尋模式

import re

text = "Contact us at support@example.com or sales@example.com"

# 找到所有電子郵件地址

    print(f"Found: {match.group()}")   # Found: 555-1234

3.6 完整範例：從多個日誌檔案中擷取資料

可執行腳本！此腳本掃描目錄中的日誌檔案， 擷取所有 ERROR 行，並寫出摘要報告。 
#!/usr/bin/env python3
"""掃描日誌檔案中的 ERROR 行並產生摘要報告。"""

import re
from pathlib import Path
from collections import defaultdict

def scan_logs(log_dir, output_file="error_report.txt"):
    """在 .log 檔案中找到所有 ERROR 行並寫出摘要。"""
    log_dir = Path(log_dir)
    errors_by_file = defaultdict(list)

    # 模式：時間戳記後接 ERROR 和訊息
    pattern = re.compile(r"(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}).*ERROR\s+(.*)")

    for log_file in sorted(log_dir.glob("*.log")):
        with open(log_file, "r") as f:
            for line_num, line in enumerate(f, 1):
                match = pattern.search(line)
                if match:
                    timestamp, message = match.groups()
                    errors_by_file[log_file.name].append(
                        (line_num, timestamp, message.strip())
                    )

    # 寫入報告
    total_errors = sum(len(errs) for errs in errors_by_file.values())

    with open(output_file, "w") as out:
        out.write(f"Error Report\n{'=' * 60}\n")
        out.write(f"Scanned: {log_dir}\n")
        out.write(f"Total errors found: {total_errors}\n\n")

        for filename, errs in sorted(errors_by_file.items()):
            out.write(f"--- {filename} ({len(errs)} errors) ---\n")
            for line_num, timestamp, message in errs:
                out.write(f"  Line {line_num} [{timestamp}]: {message}\n")
            out.write("\n")

    print(f"Report written to {output_file} ({total_errors} errors found)")

if __name__ == "__main__":
    import sys
    folder = sys.argv[1] if len(sys.argv) > 1 else "."
    scan_logs(folder)

4. 執行 Shell 命令

Python 的 subprocess 模組讓你可以在腳本中執行任何命令列 程式。它取代了較舊的 os.system() 函式（你應該避免使用）。

4.1 subprocess.run 基礎

import subprocess

# 執行簡單命令
result = subprocess.run(["echo", "Hello from Python!"])
print(f"Return code: {result.returncode}")   # 0 表示成功
為什麼用串列？將命令以串列形式傳遞 （["echo", "Hello"]）比單一字串更安全，因為 Python 會為你處理引號和跳脫字元。它也避免了 shell 注入漏洞。

4.2 擷取輸出

import subprocess

# 將 stdout 擷取為字串
result = subprocess.run(
    ["python", "--version"],
    capture_output=True,           # 擷取 stdout 和 stderr
    text=True                      # 將位元組解碼為字串
)
print(f"stdout: {result.stdout.strip()}")
print(f"stderr: {result.stderr.strip()}")
print(f"return code: {result.returncode}")

4.3 檢查回傳碼

import subprocess

# 方法 1：手動檢查
result = subprocess.run(["ls", "nonexistent_file"], capture_output=True, text=True)
if result.returncode != 0:
    print(f"Command failed: {result.stderr.strip()}")

# 方法 2：失敗時觸發例外
try:
    subprocess.run(
        ["ls", "nonexistent_file"],
        capture_output=True, text=True,
        check=True                 # 非零退出碼時觸發 CalledProcessError
    )
except subprocess.CalledProcessError as e:
    print(f"Command failed with code {e.returncode}")
    print(f"stderr: {e.stderr.strip()}")

4.4 管道命令

import subprocess

# 等同於：cat /etc/hosts | grep "localhost"
cat_proc = subprocess.run(
    ["cat", "/etc/hosts"],
    capture_output=True, text=True
)

grep_proc = subprocess.run(
    ["grep", "localhost"],
    input=cat_proc.stdout,         # 將 cat 的輸出導入 grep
    capture_output=True, text=True
)

print(grep_proc.stdout)
為什麼不用 shell=True？你可能在較舊的 程式碼中看到 subprocess.run("cat file | grep pattern", shell=True)。 雖然它能運作，但使用 shell=True 會帶來 shell 注入攻擊的風險， 並使你的腳本依賴平台。優先使用串列形式，並透過 Python 傳遞資料。

4.5 完整範例：系統健康檢查腳本

可執行腳本！一個快速的系統資訊報告器， 適用於 Linux 和 macOS。Windows 請根據需要調整命令。 
#!/usr/bin/env python3
"""快速系統健康檢查腳本。"""

import subprocess
import platform
import shutil

def run_cmd(cmd):
    """執行命令並回傳其 stdout，或錯誤訊息。"""
    try:
        result = subprocess.run(
            cmd, capture_output=True, text=True, timeout=10
        )
        return result.stdout.strip()
    except FileNotFoundError:
        return "(command not found)"
    except subprocess.TimeoutExpired:
        return "(timed out)"

def health_check():
    print("=" * 50)
    print("SYSTEM HEALTH CHECK")
    print("=" * 50)

    # 1. 基本系統資訊（跨平台）
    print(f"\nPlatform:   {platform.system()} {platform.release()}")
    print(f"Python:     {platform.python_version()}")

    # 2. 磁碟使用量（跨平台，透過 shutil）
    total, used, free = shutil.disk_usage("/")
    gb = 1024 ** 3
    print(f"\nDisk usage: {used / gb:.1f} GB used / {total / gb:.1f} GB total "
          f"({free / gb:.1f} GB free)")

    # 3. 運行時間（Linux/macOS）
    uptime = run_cmd(["uptime", "-p"])
    if uptime != "(command not found)":
        print(f"Uptime:     {uptime}")

    # 4. 記憶體資訊（Linux）
    mem = run_cmd(["free", "-h"])
    if mem != "(command not found)":
        print(f"\nMemory:\n{mem}")

    # 5. 網路連線
    ping = run_cmd(["ping", "-c", "1", "-W", "2", "8.8.8.8"])
    if "1 received" in ping or "1 packets received" in ping:
        print("\nNetwork:    Connected (ping to 8.8.8.8 OK)")
    else:
        print("\nNetwork:    Possibly offline")

    print("\n" + "=" * 50)

if __name__ == "__main__":
    health_check()

5.4 完整範例：CLI 檔案搜尋工具

可執行腳本！一個類似 grep 的小工具， 搜尋檔案中的模式並報告匹配結果。 
#!/usr/bin/env python3
"""搜尋檔案中的文字模式（類似簡易 grep）。"""

import argparse
import re
from pathlib import Path

def search_files(directory, pattern, extension, ignore_case):
    """搜尋檔案中的正規表達式模式並產生匹配結果。"""
    flags = re.IGNORECASE if ignore_case else 0
    regex = re.compile(pattern, flags)

    search_path = Path(directory)
    glob_pattern = f"*.{extension}" if extension else "*"

    for filepath in sorted(search_path.rglob(glob_pattern)):
        if not filepath.is_file():
            continue

        try:
            with open(filepath, "r", errors="ignore") as f:
                for line_num, line in enumerate(f, 1):
                    if regex.search(line):
                        yield filepath, line_num, line.rstrip()
        except (PermissionError, OSError):
            continue

def main():
    parser = argparse.ArgumentParser(
        description="Search files for a text pattern."
    )
    parser.add_argument("pattern", help="Regex pattern to search for")
    parser.add_argument("directory", nargs="?", default=".",
                        help="Directory to search (default: current)")
    parser.add_argument("-e", "--ext", default=None,
                        help="Only search files with this extension (e.g. py)")
    parser.add_argument("-i", "--ignore-case", action="store_true",
                        help="Case-insensitive search")
    args = parser.parse_args()

    count = 0
    for filepath, line_num, line in search_files(
        args.directory, args.pattern, args.ext, args.ignore_case
    ):
        # 顯示相對路徑使輸出更簡潔
        rel = filepath.relative_to(args.directory)
        print(f"{rel}:{line_num}: {line}")
        count += 1

    print(f"\n--- {count} match(es) found ---")

if __name__ == "__main__":
    main()
# 用法：
# python filesearch.py "TODO" ./src --ext py
# python filesearch.py "error" /var/log --ext log -i

6.4 完整範例：批次檔案資訊報告器

可執行腳本！掃描目錄中的圖片檔案， 報告其大小和修改日期。 
#!/usr/bin/env python3
"""報告目錄中圖片的檔案資訊（大小、日期）。"""

import os
import sys
import datetime
from pathlib import Path

IMAGE_EXTENSIONS = {".jpg", ".jpeg", ".png", ".gif", ".bmp", ".tiff", ".webp"}

def format_size(size_bytes):
    """將位元組轉換為人類可讀的字串。"""
    for unit in ("B", "KB", "MB", "GB"):
        if size_bytes < 1024:
            return f"{size_bytes:.1f} {unit}"
        size_bytes /= 1024
    return f"{size_bytes:.1f} TB"

def report_images(directory):
    """列印目錄中所有圖片檔案的報告。"""
    dir_path = Path(directory)

    if not dir_path.is_dir():
        print(f"Error: {directory} is not a valid directory")
        sys.exit(1)

    # 收集圖片檔案
    images = [
        f for f in dir_path.iterdir()
        if f.is_file() and f.suffix.lower() in IMAGE_EXTENSIONS
    ]

    if not images:
        print("No image files found.")
        return

    images.sort(key=lambda f: f.name)
    total_size = 0

    print(f"{'File':<35} {'Size':>10}  {'Modified'}")
    print("-" * 70)

    for img in images:
        stat = img.stat()
        size = stat.st_size
        total_size += size
        modified = datetime.datetime.fromtimestamp(stat.st_mtime)

        print(f"{img.name:<35} {format_size(size):>10}  {modified:%Y-%m-%d %H:%M}")

    print("-" * 70)
    print(f"{'Total: ' + str(len(images)) + ' files':<35} {format_size(total_size):>10}")

if __name__ == "__main__":
    target = sys.argv[1] if len(sys.argv) > 1 else "."
    report_images(target)
# 用法：
# python image_report.py ~/Pictures
#
# File                                    Size  Modified
# ----------------------------------------------------------------------
# beach_sunset.jpg                      2.3 MB  2025-06-15 14:30
# diagram.png                         145.2 KB  2025-07-01 09:12
# screenshot_2025.png                  890.4 KB  2025-08-20 11:45
# ----------------------------------------------------------------------
# Total: 3 files                        3.3 MB

7. 路徑處理

正確的路徑處理對於跨平台腳本至關重要。Python 提供兩種方法： 傳統的 os.path 模組和現代的 pathlib 模組。

7.1 os.path 與 pathlib 比較

操作 os.path pathlib
合併路徑 os.path.join("a", "b", "c.txt") Path("a") / "b" / "c.txt"
取得檔名 os.path.basename(p) Path(p).name
取得目錄 os.path.dirname(p) Path(p).parent
取得副檔名 os.path.splitext(p)[1] Path(p).suffix
取得檔名（無副檔名） os.path.splitext(os.path.basename(p))[0] Path(p).stem
檔案是否存在？ os.path.exists(p) Path(p).exists()
是否為檔案？ os.path.isfile(p) Path(p).is_file()
是否為目錄？ os.path.isdir(p) Path(p).is_dir()
絕對路徑 os.path.abspath(p) Path(p).resolve()
家目錄 os.path.expanduser("~") Path.home()
目前目錄 os.getcwd() Path.cwd()
應該使用哪一個？新程式碼優先使用 pathlib。 它更易讀（/ 運算子很直覺）、回傳豐富的物件而非原始字串， 並自動處理平台差異。你在較舊的程式碼中會遇到 os.path， 所以兩者都值得了解。

8.1 監看資料夾中的新檔案

#!/usr/bin/env python3
"""輪詢目錄，當新檔案出現時做出反應。"""

import time
from pathlib import Path

def watch_folder(folder, interval=2):
    """監看資料夾，當新檔案到達時列印訊息。"""
    folder = Path(folder)
    known_files = set(folder.iterdir())

    print(f"Watching {folder} for new files (Ctrl+C to stop)...")

    try:
        while True:
            time.sleep(interval)
            current_files = set(folder.iterdir())
            new_files = current_files - known_files

            for f in new_files:
                if f.is_file():
                    print(f"  New file detected: {f.name} "
                          f"({f.stat().st_size} bytes)")

            known_files = current_files
    except KeyboardInterrupt:
        print("\nStopped watching.")

if __name__ == "__main__":
    import sys
    target = sys.argv[1] if len(sys.argv) > 1 else "."
    watch_folder(target)

8.2 備份腳本

#!/usr/bin/env python3
"""建立目錄的時間戳記備份。"""

import shutil
import datetime
from pathlib import Path

def backup_directory(source, backup_root="backups"):
    """將來源目錄複製到帶有時間戳記的 backup_root 中。"""
    source = Path(source)
    backup_root = Path(backup_root)

    if not source.is_dir():
        print(f"Error: {source} is not a directory")
        return

    # 建立基於時間戳記的資料夾名稱
    timestamp = datetime.datetime.now().strftime("%Y%m%d_%H%M%S")
    dest = backup_root / f"{source.name}_{timestamp}"

    print(f"Backing up {source} -> {dest}")
    shutil.copytree(source, dest)

    # 計算複製了什麼
    file_count = sum(1 for _ in dest.rglob("*") if _.is_file())
    print(f"Done! {file_count} file(s) backed up.")

if __name__ == "__main__":
    import sys
    if len(sys.argv) < 2:
        print("Usage: python backup.py  [backup_root]")
        sys.exit(1)
    src = sys.argv[1]
    dst = sys.argv[2] if len(sys.argv) > 2 else "backups"
    backup_directory(src, dst)

8.3 跨檔案尋找與取代

#!/usr/bin/env python3
"""在所有匹配模式的檔案中尋找和取代字串。"""

from pathlib import Path

def find_and_replace(directory, glob_pattern, old_text, new_text, dry_run=True):
    """在匹配 glob_pattern 的檔案中將 old_text 取代為 new_text。"""
    directory = Path(directory)
    changed_files = 0

    for filepath in sorted(directory.rglob(glob_pattern)):
        if not filepath.is_file():
            continue

        try:
            content = filepath.read_text()
        except (UnicodeDecodeError, PermissionError):
            continue

        if old_text not in content:
            continue

        count = content.count(old_text)

        if dry_run:
            print(f"  [DRY RUN] {filepath}: {count} occurrence(s)")
        else:
            new_content = content.replace(old_text, new_text)
            filepath.write_text(new_content)
            print(f"  Updated {filepath}: {count} replacement(s)")

        changed_files += 1

    mode = "DRY RUN" if dry_run else "DONE"
    print(f"\n[{mode}] {changed_files} file(s) affected.")

if __name__ == "__main__":
    import sys
    if len(sys.argv) < 4:
        print("Usage: python replace.py    [--apply]")
        print("  Without --apply, runs in dry-run mode (shows what would change)")
        sys.exit(1)

    apply = "--apply" in sys.argv
    find_and_replace(
        directory=sys.argv[1],
        glob_pattern="*.py",
        old_text=sys.argv[2],
        new_text=sys.argv[3],
        dry_run=not apply
    )
安全第一：注意 dry_run=True 預設值。 在進行實際修改之前，務必預覽大量編輯腳本會更改的內容。 這個模式 — 預設為試執行，--apply 旗標才執行 — 可以防止意外。

8.4 從資料產生報告

#!/usr/bin/env python3
"""讀取 CSV 資料檔案並產生摘要報告。"""

import csv
from pathlib import Path
from collections import defaultdict

def generate_report(data_dir, output_file="summary_report.txt"):
    """讀取所有 CSV 檔案並產生摘要。"""
    data_dir = Path(data_dir)
    all_stats = []

    for csv_file in sorted(data_dir.glob("*.csv")):
        with open(csv_file, "r", newline="") as f:
            reader = csv.DictReader(f)
            rows = list(reader)

        if not rows:
            continue

        # 收集基本統計
        stats = {
            "file": csv_file.name,
            "rows": len(rows),
            "columns": len(rows[0]),
            "column_names": list(rows[0].keys()),
        }
        all_stats.append(stats)

    # 寫入報告
    with open(output_file, "w") as out:
        out.write("DATA SUMMARY REPORT\n")
        out.write("=" * 50 + "\n\n")

        for s in all_stats:
            out.write(f"File: {s['file']}\n")
            out.write(f"  Rows: {s['rows']}\n")
            out.write(f"  Columns ({s['columns']}): {', '.join(s['column_names'])}\n\n")

        out.write(f"Total files processed: {len(all_stats)}\n")
        out.write(f"Total rows across all files: "
                  f"{sum(s['rows'] for s in all_stats)}\n")

    print(f"Report written to {output_file}")

if __name__ == "__main__":
    import sys
    folder = sys.argv[1] if len(sys.argv) > 1 else "."
    generate_report(folder)

8.5 清理暫存檔案

#!/usr/bin/env python3
"""從專案目錄中刪除暫存和快取檔案。"""

import os
from pathlib import Path

# 要清理的模式
JUNK_PATTERNS = [
    "**/*.pyc",              # 編譯的 Python 位元碼
    "**/__pycache__",        # 位元碼快取目錄
    "**/*.tmp",              # 暫存檔案
    "**/*.log",              # 日誌檔案
    "**/.DS_Store",          # macOS 資料夾中繼資料
    "**/Thumbs.db",          # Windows 縮圖快取
]

def cleanup(directory, dry_run=True):
    """移除匹配 JUNK_PATTERNS 的暫存檔案和目錄。"""
    directory = Path(directory)
    removed_files = 0
    removed_dirs = 0
    freed_bytes = 0

    for pattern in JUNK_PATTERNS:
        for item in directory.glob(pattern):
            if item.is_file():
                size = item.stat().st_size
                if dry_run:
                    print(f"  [DEL FILE] {item} ({size:,} bytes)")
                else:
                    item.unlink()
                    print(f"  Deleted: {item}")
                removed_files += 1
                freed_bytes += size
            elif item.is_dir():
                if dry_run:
                    print(f"  [DEL DIR]  {item}/")
                else:
                    import shutil
                    shutil.rmtree(item)
                    print(f"  Removed: {item}/")
                removed_dirs += 1

    mode = "DRY RUN" if dry_run else "CLEANED"
    print(f"\n[{mode}] {removed_files} file(s), {removed_dirs} dir(s), "
          f"~{freed_bytes / 1024:.1f} KB freed")

if __name__ == "__main__":
    import sys
    target = sys.argv[1] if len(sys.argv) > 1 else "."
    apply = "--apply" in sys.argv
    cleanup(target, dry_run=not apply)
# 用法：
# python cleanup.py ./my_project              （預覽將被刪除的內容）
# python cleanup.py ./my_project --apply      （實際刪除）

  • Python 的標準函式庫提供了你所需的一切：檔案管理、文字處理、shell 命令和 CLI 工具 — 不需要第三方套件。
