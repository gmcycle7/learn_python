Python 速查表

這是一份精簡、可快速掃描的 Python 日常快速參考。收藏本頁或列印出來 — 每個範例都控制在一到兩行，讓你一眼就能找到所需內容。

1. 基本語法

變數與賦值

x = 10
name = "Alice"

輸出

print("Hello, world!")
print(f"Name: {name}, Age: {x}")

輸入

answer = input("Enter your name: ")

註解

# 這是單行註解
x = 5  # 行內註解

縮排

if True:
    print("縮排定義程式碼區塊 — 使用4個空格")

2. 資料型別

x = 42              # 整數 int
pi = 3.14           # 浮點數 float
name = "Alice"      # 字串 str
flag = True         # 布林值 bool
nums = [1, 2, 3]    # 串列 list
pair = (1, 2)       # 元組 tuple
info = {"a": 1}     # 字典 dict
unique = {1, 2, 3}  # 集合 set
empty = None        # NoneType

檢查型別

type(x)        # <class 'int'>
isinstance(x, int)  # True

3. 字串操作

建立與串接

s = "hello"
full = "hello" + " " + "world"

F-字串（格式化字串）

name = "Alice"
print(f"Hi, {name}! 2+2={2+2}")

切片

s = "hello"
s[0]     # 'h'
s[1:4]   # 'ell'
s[::-1]  # 'olleh'（反轉）

常用方法

"hello".upper()                # 'HELLO'
"HELLO".lower()                # 'hello'
"  hi  ".strip()               # 'hi'
"a,b,c".split(",")             # ['a', 'b', 'c']
", ".join(["a", "b", "c"])     # 'a, b, c'
"hello".replace("l", "L")     # 'heLLo'
"hello".find("ll")            # 2（找不到回傳 -1）
"hello".count("l")            # 2
"hello".startswith("he")      # True
"hello".endswith("lo")        # True
"ell" in "hello"              # True

4. 串列操作

建立與存取

nums = [10, 20, 30, 40]
nums[0]     # 10
nums[-1]    # 40
nums[1:3]   # [20, 30]

修改

nums.append(50)        # [10, 20, 30, 40, 50]
nums.extend([60, 70])  # 在末尾添加 60, 70
nums.insert(0, 5)      # 在索引 0 處插入 5
nums.remove(20)        # 移除第一個出現的 20
nums.pop()             # 移除並回傳最後一個元素
nums.pop(0)            # 移除並回傳索引 0 的元素

排序與反轉

nums.sort()            # 原地排序（升序）
nums.sort(reverse=True)
nums.reverse()         # 原地反轉
sorted(nums)           # 回傳新的排序串列

長度與成員檢查

len(nums)          # 元素數量
30 in nums         # True

串列推導式

squares = [x**2 for x in range(5)]       # [0, 1, 4, 9, 16]
evens = [x for x in range(10) if x % 2 == 0]

5. 字典操作

建立與存取

d = {"name": "Alice", "age": 30}
d["name"]          # 'Alice'
d.get("age")       # 30
d.get("x", 0)      # 0（鍵不存在時的預設值）

新增、更新與刪除

d["email"] = "a@b.com"   # 新增 / 更新
d.update({"age": 31, "city": "NY"})
del d["email"]           # 刪除鍵

迭代

d.keys()           # dict_keys(['name', 'age'])
d.values()         # dict_values(['Alice', 30])
d.items()          # dict_items([('name','Alice'), ('age',30)])

成員檢查與推導式

"name" in d              # True（檢查鍵）
squares = {x: x**2 for x in range(5)}

6. 集合操作

s = {1, 2, 3}
s.add(4)               # {1, 2, 3, 4}
s.remove(2)            # {1, 3, 4} — 元素不存在時會報錯
s.discard(99)          # 元素不存在時不會報錯

a = {1, 2, 3}
b = {3, 4, 5}
a | b                  # {1, 2, 3, 4, 5}  聯集
a & b                  # {3}              交集
a - b                  # {1, 2}           差集
3 in a                 # True

7. 元組操作

t = (1, 2, 3)
t[0]                   # 1
a, b, c = t            # 拆解：a=1, b=2, c=3
single = (42,)         # 單元素元組（注意逗號）

8. 流程控制

if / elif / else

if x > 0:
    print("positive")
elif x == 0:
    print("zero")
else:
    print("negative")

for 迴圈

for item in [1, 2, 3]:
    print(item)

while 迴圈

while x > 0:
    x -= 1

break、continue、pass

for i in range(10):
    if i == 5: break       # 跳出迴圈
    if i % 2 == 0: continue  # 跳到下一次迭代
    pass                   # 不做任何事（佔位符）

range、enumerate、zip

range(5)             # 0, 1, 2, 3, 4
range(2, 8)          # 2, 3, 4, 5, 6, 7
range(0, 10, 2)      # 0, 2, 4, 6, 8

for i, val in enumerate(["a", "b", "c"]):
    print(i, val)    # 0 a, 1 b, 2 c

for a, b in zip([1,2], ["x","y"]):
    print(a, b)      # 1 x, 2 y

9. 函式

基本函式

def greet(name):
    return f"Hello, {name}!"

預設參數

def greet(name, greeting="Hi"):
    return f"{greeting}, {name}!"

*args 和 **kwargs

def add(*args):
    return sum(args)           # add(1, 2, 3) → 6

def info(**kwargs):
    for k, v in kwargs.items():
        print(f"{k}: {v}")     # info(name="Alice", age=30)

Lambda（匿名函式）

double = lambda x: x * 2     # double(5) → 10
sorted(names, key=lambda s: s.lower())

10. 檔案處理

讀取

with open("file.txt", "r") as f:
    content = f.read()         # 整個檔案作為字串

with open("file.txt", "r") as f:
    lines = f.readlines()      # 行的串列

寫入與附加

with open("file.txt", "w") as f:
    f.write("Hello\n")        # 覆寫

with open("file.txt", "a") as f:
    f.write("More text\n")    # 附加

CSV

import csv
with open("data.csv") as f:
    reader = csv.reader(f)
    for row in reader:
        print(row)             # 每一行是一個串列

11. 錯誤處理

try / except / else / finally

try:
    result = 10 / x
except ZeroDivisionError:
    print("Cannot divide by zero")
except (TypeError, ValueError) as e:
    print(f"Error: {e}")
else:
    print("Success")          # 沒有例外時執行
finally:
    print("Always runs")

常見例外

ValueError       # 值錯誤（例如 int("abc")）
TypeError        # 型別錯誤（例如 "a" + 1）
KeyError         # 字典鍵不存在
IndexError       # 串列索引超出範圍
FileNotFoundError
ZeroDivisionError
AttributeError   # 物件沒有該屬性
ImportError      # 找不到模組
NameError        # 未定義的變數

14. 運算子參考

算術運算

x + y       # 加法
x - y       # 減法
x * y       # 乘法
x / y       # 除法（浮點數）
x // y      # 整數除法
x % y       # 取餘數

比較運算

x == y      # 等於
x != y      # 不等於
x < y       # 小於
x > y       # 大於
x <= y      # 小於或等於

邏輯運算
