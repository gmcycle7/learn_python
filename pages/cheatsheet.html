Python Cheat Sheet

A dense, scannable quick-reference for everyday Python. Bookmark this page or print it out — every example is kept to one or two lines so you can find what you need at a glance.

1. Basic Syntax

Variables & Assignment

x = 10
name = "Alice"

Print

print("Hello, world!")
print(f"Name: {name}, Age: {x}")

Input

answer = input("Enter your name: ")

Comments

# This is a single-line comment
x = 5  # inline comment

Indentation

if True:
    print("Indentation defines blocks — use 4 spaces")

2. Data Types

x = 42              # int
pi = 3.14           # float
name = "Alice"      # str
flag = True         # bool
nums = [1, 2, 3]    # list
pair = (1, 2)       # tuple
info = {"a": 1}     # dict
unique = {1, 2, 3}  # set
empty = None        # NoneType

Check Type

type(x)        # <class 'int'>
isinstance(x, int)  # True

3. String Operations

Create & Concatenate

s = "hello"
full = "hello" + " " + "world"

F-String (Formatted String)

name = "Alice"
print(f"Hi, {name}! 2+2={2+2}")

Slicing

s = "hello"
s[0]     # 'h'
s[1:4]   # 'ell'
s[::-1]  # 'olleh' (reverse)

Common Methods

"hello".upper()                # 'HELLO'
"HELLO".lower()                # 'hello'
"  hi  ".strip()               # 'hi'
"a,b,c".split(",")             # ['a', 'b', 'c']
", ".join(["a", "b", "c"])     # 'a, b, c'
"hello".replace("l", "L")     # 'heLLo'
"hello".find("ll")            # 2 (-1 if not found)
"hello".count("l")            # 2
"hello".startswith("he")      # True
"hello".endswith("lo")        # True
"ell" in "hello"              # True

4. List Operations

Create & Access

nums = [10, 20, 30, 40]
nums[0]     # 10
nums[-1]    # 40
nums[1:3]   # [20, 30]

Modify

nums.append(50)        # [10, 20, 30, 40, 50]
nums.extend([60, 70])  # adds 60, 70 to end
nums.insert(0, 5)      # insert 5 at index 0
nums.remove(20)        # remove first occurrence of 20
nums.pop()             # remove & return last item
nums.pop(0)            # remove & return item at index 0

Sort & Reverse

nums.sort()            # sort in place (ascending)
nums.sort(reverse=True)
nums.reverse()         # reverse in place
sorted(nums)           # returns new sorted list

Length & Membership

len(nums)          # number of items
30 in nums         # True

List Comprehension

squares = [x**2 for x in range(5)]       # [0, 1, 4, 9, 16]
evens = [x for x in range(10) if x % 2 == 0]

5. Dictionary Operations

Create & Access

d = {"name": "Alice", "age": 30}
d["name"]          # 'Alice'
d.get("age")       # 30
d.get("x", 0)      # 0 (default if key missing)

Add, Update & Delete

d["email"] = "a@b.com"   # add / update
d.update({"age": 31, "city": "NY"})
del d["email"]           # delete key

Iterate

d.keys()           # dict_keys(['name', 'age'])
d.values()         # dict_values(['Alice', 30])
d.items()          # dict_items([('name','Alice'), ('age',30)])

Membership & Comprehension

"name" in d              # True (checks keys)
squares = {x: x**2 for x in range(5)}

6. Set Operations

s = {1, 2, 3}
s.add(4)               # {1, 2, 3, 4}
s.remove(2)            # {1, 3, 4} — error if missing
s.discard(99)          # no error if missing

a = {1, 2, 3}
b = {3, 4, 5}
a | b                  # {1, 2, 3, 4, 5}  union
a & b                  # {3}              intersection
a - b                  # {1, 2}           difference
3 in a                 # True

7. Tuple Operations

t = (1, 2, 3)
t[0]                   # 1
a, b, c = t            # unpacking: a=1, b=2, c=3
single = (42,)         # single-element tuple (note the comma)

8. Control Flow

if / elif / else

if x > 0:
    print("positive")
elif x == 0:
    print("zero")
else:
    print("negative")

for Loop

for item in [1, 2, 3]:
    print(item)

while Loop

while x > 0:
    x -= 1

break, continue, pass

for i in range(10):
    if i == 5: break       # exit loop
    if i % 2 == 0: continue  # skip to next iteration
    pass                   # do nothing (placeholder)

range, enumerate, zip

range(5)             # 0, 1, 2, 3, 4
range(2, 8)          # 2, 3, 4, 5, 6, 7
range(0, 10, 2)      # 0, 2, 4, 6, 8

for i, val in enumerate(["a", "b", "c"]):
    print(i, val)    # 0 a, 1 b, 2 c

for a, b in zip([1,2], ["x","y"]):
    print(a, b)      # 1 x, 2 y

9. Functions

Basic Function

def greet(name):
    return f"Hello, {name}!"

Default Arguments

def greet(name, greeting="Hi"):
    return f"{greeting}, {name}!"

*args and **kwargs

def add(*args):
    return sum(args)           # add(1, 2, 3) → 6

def info(**kwargs):
    for k, v in kwargs.items():
        print(f"{k}: {v}")     # info(name="Alice", age=30)

Lambda (Anonymous Function)

double = lambda x: x * 2     # double(5) → 10
sorted(names, key=lambda s: s.lower())

10. File Handling

Read

with open("file.txt", "r") as f:
    content = f.read()         # entire file as string

with open("file.txt", "r") as f:
    lines = f.readlines()      # list of lines

Write & Append

with open("file.txt", "w") as f:
    f.write("Hello\n")        # overwrite

with open("file.txt", "a") as f:
    f.write("More text\n")    # append

CSV

import csv
with open("data.csv") as f:
    reader = csv.reader(f)
    for row in reader:
        print(row)             # each row is a list

11. Error Handling

try / except / else / finally

try:
    result = 10 / x
except ZeroDivisionError:
    print("Cannot divide by zero")
except (TypeError, ValueError) as e:
    print(f"Error: {e}")
else:
    print("Success")          # runs if no exception
finally:
    print("Always runs")

Common Exceptions

ValueError       # wrong value (e.g. int("abc"))
TypeError        # wrong type (e.g. "a" + 1)
KeyError         # missing dict key
IndexError       # list index out of range
FileNotFoundError
ZeroDivisionError
AttributeError   # object has no such attribute
ImportError      # module not found
NameError        # undefined variable

12. Common Imports

import os                  # file paths, env vars, system commands
import sys                 # interpreter settings, argv
import math                # sqrt, ceil, floor, pi, e
import random              # randint, choice, shuffle
import datetime            # date, time, timedelta
import json                # json.loads(), json.dumps()
import csv                 # csv.reader(), csv.writer()
from pathlib import Path   # modern file paths
import re                  # regular expressions
import shutil              # copy/move files and dirs
import glob                # file pattern matching
import subprocess          # run shell commands
from collections import Counter, defaultdict

Quick Examples

os.getcwd()                        # current directory
os.listdir(".")                    # list files
Path("data/file.txt").read_text()  # read file
math.sqrt(16)                      # 4.0
random.randint(1, 10)              # random int 1-10
json.loads('{"a": 1}')             # parse JSON string
datetime.datetime.now()            # current date/time
re.findall(r"\d+", "abc123")      # ['123']

13. Package Management

pip install requests           # install a package
pip install requests==2.28.0   # specific version
pip list                       # list installed packages
pip freeze > requirements.txt  # save dependencies
pip install -r requirements.txt  # install from file

Virtual Environments

python -m venv myenv           # create virtual env
source myenv/bin/activate      # activate (Linux/Mac)
myenv\Scripts\activate         # activate (Windows)
deactivate                     # deactivate

14. Operators Reference

Arithmetic

x + y       # addition
x - y       # subtraction
x * y       # multiplication
x / y       # division (float)
x // y      # floor division (int)
x % y       # modulo (remainder)
x ** y      # exponentiation

Comparison

x == y      # equal
x != y      # not equal
x < y       # less than
x > y       # greater than
x <= y      # less than or equal
x >= y      # greater than or equal

Logical

x and y     # True if both True
x or y      # True if either True
not x       # invert boolean

Assignment

x = 10
x += 5      # x = x + 5
x -= 3      # x = x - 3
x *= 2      # x = x * 2
x /= 4      # x = x / 4

Membership

"a" in "abc"          # True
3 not in [1, 2]       # True

Identity

x is None             # True if x is None
x is not None         # True if x is not None
